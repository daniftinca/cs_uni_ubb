CMMDC

   function cmmdc(a, b) {
    var aux = 0;
    if(a===0){
        return b;
    }
    if(b===0){
        return a;
    }
    while (b) {
        aux = a;
        a = b;
        b = aux % b;
    }
    return a
}

EBNF

    functie::=antet"{"listai"}"
    antet::=function ID"("[listapformali]")"
    listap::=par["," par]
    listapformali::= parFormal["," par]
    parFormal::= ID
    par::= ID|CONST
    listai::=instr | instr listai
    instr::= declarare | atribuire | apelsingur | while | return | if | ifelse | functie
    apel::= ID"("[listap]")" | ID.ID"("[listap]")"
    apelsingur::=apel";"
    while::= "while" "("cond")""{"listai"}"
    cond::= expr ("<"|">"|"=="|"<="|">="|"===") expr
    if::= "if" "(" cond ")" "{" listai "}"
    else::= "else" "{"listai"}"
    elif::== "else " if | "else " if elif
    ifelse::= if [elif] else | if else
    return::= "return "expr";"
    declarare::= var ID";" | var atribuire
    atribuire::= ID"="expr";"
    expr::= ID|CONST|apel|listaElem|expr("+"|"-"|"%"|"/"|"*")expr
    listaElem::="["expr[","expr]"]"
    defclasa::="class "ID"{"listai"}"
    instrClasa::= declarare | metoda
    metoda::= ID"("[listapFormali]")""{"[listai]"}
